\section{Variational Inequality}

Inside the supremum: \\

\scalebox{0.9}{$P_t^X + S_t P_t^Y + R_t = \displaystyle\int_0^t \left( \varphi_c(Y_{s^-} - \xi) - \varphi_c(Y_{s^-}) - \xi S_t + r(Y_{s^-}, Z_{s^-}) \right) dN_s^a + \displaystyle\int_0^t \left( \varphi_c(Y_{s^-} + \xi) - \varphi_c(Y_{s^-}) + \xi S_t + r(Y_{s^-}, Z_{s^-}) \right) dN_s^b$}\\

Value function, at time $t$ for a stopping time $\tau$ in $[t,T]$: \\

\scalebox{0.9}{$v(t,y,z,S) = \sup_{\tau \in \Tc_{t,T}} \mathbb{E} \left[ \displaystyle\int_{t}^{\tau} \beta^a_\tau(Y_{s^-}, Z_{s^-}) \, dN^a_s + \displaystyle\int_{t}^{\tau} \beta^b_\tau(Y_{s^-}, Z_{s^-}) \, dN^b_s \right]$}\\

where:\\

\scalebox{0.9}{$\beta^a_\tau(Y_{s^-}, Z_{s^-}) = \varphi_c(Y_{s^-} - \xi) - \varphi_c(Y_{s^-}) - \xi S_\tau + r(Y_{s^-}, Z_{s^-})$}

\scalebox{0.9}{$\beta^b_\tau(Y_{s^-}, Z_{s^-}) = \varphi_c(Y_{s^-} + \xi) - \varphi_c(Y_{s^-}) + \xi S_\tau + r(Y_{s^-}, Z_{s^-})$}\\

Dynamic principle, we stop at time $t$ ($\tau=t$) or we continue ($\tau>t$): \\

\scalebox{0.9}{$v(t,y,z,S) = \max \left \{ \mathbb{E} \left[ \displaystyle\int_{t}^{t+h} \beta^b_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^b + \displaystyle\int_{t}^{t+h} \beta^a_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^a + v(t+h, Y_{\tau}^{t, Y^{t,y}_{t+h}}, Z_{\tau}^{t, Z^{t,z}_{t+h}}, S_{\tau}^{t, S^{t,S}_{t+h}}) \Bigm| \mathcal{F}_t \right],0 \right \}$}

\scalebox{0.9}{$0=\max \left \{\mathbb{E} \left[ \displaystyle\int_{t}^{t+h} \beta^b_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^b + \displaystyle\int_{t}^{t+h} \beta^a_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^a + v(t+h, Y_{\tau}^{t, Y^{t,y}_{t+h}}, Z_{\tau}^{t, Z^{t,z}_{t+h}}, S_{\tau}^{t, S^{t,S}_{t+h}}) - v(t, Y_{\tau}^{t,y}, Z_{\tau}^{t,z}, S_{\tau}^{t,s}) \Bigm| \mathcal{F}_t \right],-v(t,y,z,S) \right \}$}

\scalebox{0.9}{$0=\max \left \{\mathbb{E} \left[ \displaystyle\int_{t}^{t+h} \beta^b_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^b + \displaystyle\int_{t}^{t+h} \beta^a_{t+h}(Y_{s^-}, Z_{s^-}) \, dN_s^a + \displaystyle\int_{t}^{t+h} \, dv(u, Y_u, Z_u, S_u) \Bigm| \mathcal{F}_t \right],-v(t,y,z,S) \right \}$}\\

Applying ItĂ´ formula on $v(t,Y_t,Z_t,S_t)$ with jump and diffusion processes:\\

\scalebox{0.9}{%
$\begin{aligned}
0 = \max \bigg\{ &\mathbb{E} \bigg[ \displaystyle\int_{t}^{t+h} \frac{\partial}{\partial t} v(u, Y_u, Z_u, S_u) \, du + \displaystyle\int_{t}^{t+h} \frac{1}{2} \sigma^2 \frac{\partial^2}{\partial S^2} v(u, Y_u, Z_u, S_u) \, du + \displaystyle\int_{t}^{t+h} \sigma \frac{\partial}{\partial S} v(u, Y_u, Z_u, S_u) \, dW_u \\
&\quad + \displaystyle\int_{t}^{t+h} \big( \beta^a_{t+h}(Y_{u^-}, Z_{u^-}) + v(u, Y_{u^-} - \xi, Z_{u^-} - \varphi'_c(Y_{u^-} - \xi) + \varphi'_c(Y_{u^-}), S_u) - v(u, Y_{u^-}, Z_{u^-}, S_u) \big) \, dN_u^a \\
&\quad + \displaystyle\int_{t}^{t+h} \big( \beta^b_{t+h}(Y_{u^-}, Z_{u^-}) + v(u, Y_{u^-} + \xi, Z_{u^-} - \varphi'_c(Y_{u^-} + \xi) + \varphi'_c(Y_{u^-}), S_u) - v(u, Y_{u^-}, Z_{u^-}, S_u) \big) \, dN_u^b \Bigm| \mathcal{F}_t \bigg] \notag, -v(t, y, z, S) \bigg\}
\end{aligned}$
}\\

Recalling that $d\tilde{N}_t^a = dN_t^a - \lambda_t^a \, dt$, $d\tilde{N}_t^b = dN_t^b - \lambda_t^a \, dt$ with $\lambda_t^a = \bar{\lambda}^a(Z_{t^-}, Y_{t^-}, S_t)$, $\bar{\lambda}^b(Z_{t^-}, Y_{t^-}, S_t)$.\\

Expectation of martingale parts $\displaystyle\int_{}^{}.dW_t$ and $\displaystyle\int_{}^{}.d\tilde{N}_t$ are 0.\\

\scalebox{0.9}{%
$\begin{aligned}
0 = \max \bigg\{ & \displaystyle\int_{t}^{t+h} \frac{\partial}{\partial t} v(u, y, z, S) \, + \frac{1}{2} \sigma^2 \frac{\partial^2}{\partial S^2} v(u, y, z, S) \, \\
&\quad + \bar{\lambda}^a(z, y, S) \big[ \varphi_c(y - \xi) - \varphi_c(y) - \xi S_{t+h} + r(y, z) + v(u, y + \xi, z - \varphi'_c(y + \xi) + \varphi'_c(y), S) - v(u, y, z, S) \big] \,  \\
&\quad + \bar{\lambda}^b(z, y, S) \big[ \varphi_c(y + \xi) - \varphi_c(y) + \xi S_{t+h} + r(y, z) + v(u, y - \xi, z - \varphi'_c(y- \xi) + \varphi'_c(y), S) - v(u, y, z, S) \big] \, du, -v(t, y, z, S) \bigg\}
\end{aligned}$
}\\

Therefore, for $h$ small enough, the variational inequality is:\\

\scalebox{0.9}{%
$\begin{aligned}
0 = \max \bigg\{ & \frac{\partial}{\partial t} v(t, y, z, S) \, + \frac{1}{2} \sigma^2 \frac{\partial^2}{\partial S^2} v(t, y, z, S) \, \\
&\quad + \bar{\lambda}^a(z, y, S) \big[ \varphi_c(y - \xi) - \varphi_c(y) - \xi S_t + r(y, z) + v(t, y + \xi, z - \varphi'_c(y + \xi) + \varphi'_c(y), S) - v(t, y, z, S) \big] \,  \\
&\quad + \bar{\lambda}^b(z, y, S) \big[ \varphi_c(y + \xi) - \varphi_c(y) + \xi S_t + r(y, z) + v(t, y - \xi, z - \varphi'_c(y- \xi) + \varphi'_c(y), S) - v(t, y, z, S) \big] \, , -v(t, y, z, S) \bigg\}
\end{aligned}$
}\\

Tentative with risk-averse framework:

With terminal condition : $v(T,y,z,S)=- \exp \left\{ - \gamma \left( x-X_{0} +(y-Y_{0})S + r(y,z) \right) \right\}$ \\

\section{Euler Scheme}
~
We have $x,y,z$ variables linked, $\frac{x}{y}=z$ and $\varphi_{c}(y) = \frac{c}{y} = x$. Therefore, the problem can be reduce from dimension 4 to dimension 3, we write again the EDP in the variational inequality.\\

\scalebox{1}{%
$\begin{aligned}
0 = & \frac{\partial}{\partial t} v(t, y, S) \, + \frac{1}{2} \sigma^2 \frac{\partial^2}{\partial S^2} v(t, y, S) \, \\
&\quad + \bar{\lambda}^a(y, S) \big[ \varphi_c \big(y - \xi \big) - \varphi_c \big(y \big) - \xi S_t + r \big(\frac{c}{y^2} \big) + v(t, y + \xi, S) - v(t, y, S) \big] \,  \\
&\quad + \bar{\lambda}^b(y, S) \big[ \varphi_c \big(y + \xi \big) - \varphi_c \big(y \big) + \xi S_t + r \big(\frac{c}{y^2} \big) + v(t, y - \xi, S) - v(t, y, S) \big] \
\end{aligned}$
}\\

Backward Euler discretization for $\delta$ and $h$ small enough and $\xi$ for the jumps:\\

Diffusion part for $i \in \{0,...,I-1\}$, $t_0=0$ and $t_I=T$ and $j \in \{1,...,J-1\}$:\\

$\begin{aligned}
&\frac{\partial}{\partial t} v(t, y, S) \approx \frac{v(t + \delta,y,S) - v(t,y,S)}{\delta} = \frac{v(t_{i+1},y_l,S_j) - v(t_{i},y_l,S_j)}{\delta} \\
&\frac{\partial^2}{\partial S^2} v(t, y, S) \approx \frac{v(t,y,S+h) - 2v(t,y,S) + v(t,y,S-h)}{h^2} = \frac{v(t_{i},y_l,S_{j+1}) - 2v(t_{i},y_l,S_j) + v(t_{i},y_l,S_{j-1})}{h^2}
\end{aligned}$\\ \\

Jump part for $l \in \{1,...,L-1\}$:\\

$\begin{aligned}
&\bar{\lambda}^a(y_l, S_j) \big[ \varphi_c \big(y_{l-1} \big) - \varphi_c \big(y_l \big) - \xi S_j + r \big(\frac{c}{y_l^2} \big) + v(t_{i-1}, y_{l+1}, S_j) - v(t_{i-1}, y_l, S_j) \big] \\
&\bar{\lambda}^b(y_l, S_j) \big[ \varphi_c \big(y_{l+1} \big) - \varphi_c \big(y_l \big) + \xi S_j + r \big(\frac{c}{y_l^2} \big) + v(t_{i-1}, y_{l-1}, S_j) - v(t_{i-1}, y_l, S_j) \big]
\end{aligned}$ \\

\underline{Full Euler scheme:}\\

\scalebox{1}{%
$\begin{aligned}
v(t_{i+1},y_l,S_j) = & ~ v(t_{i},y_l,S_j) \big [ 1 + \delta \big(\frac{\sigma^2}{h^2} + \bar{\lambda}^a(y_l, S_j) + \bar{\lambda}^b(y_l, S_j)\big) \big] \, \\
&+ v(t_{i},y_l,S_{j+1}) \big[-\frac{\delta \sigma^2}{2h^2} \big] \, \\
&+ v(t_{i},y_l,S_{j-1}) \big[-\frac{\delta \sigma^2}{2h^2} \big] \, \\
&+ v(t_{i}, y_{l+1}, S_j) \big[-\bar{\lambda}^a(y_l, S_j) \big] \, \\
&+ v(t_{i}, y_{l-1}, S_j) \big[-\bar{\lambda}^b(y_l, S_j) \big] \, \\
&- \bar{\lambda}^a(y_l, S_j) \big[ \varphi_c \big(y_{l-1} \big) - \varphi_c \big(y_l \big) - \xi S_j + r \big(\frac{c}{y_l^2} \big) \big] - \bar{\lambda}^b(y_l, S_j) \big[ \varphi_c \big(y_{l+1} \big) - \varphi_c \big(y_l \big) + \xi S_j + r \big(\frac{c}{y_l^2} \big) \big] \
\end{aligned}$
}\\

Operator Splitting Idea: At each time of the discretization, solve the linear system step without jumps and then adjust with jump parts.\\

\underline{Boundary conditions:}\\

A terminal time $T$, we have $v(T,y,S)=0$, in the Euler scheme this means:\\

$\begin{aligned}
&\forall \, l \in \{0, \ldots, L\}, \, j \in \{0, \ldots, J\} , v(t_I, y_l, S_j) = 0\\
\end{aligned}$\\

Neumann condition:
$\left. \frac{\partial^2 v(t, y, S)}{\partial S^2} \right|_{S = S_0} = 0, \quad 
\left. \frac{\partial^2 v(t, y, S)}{\partial S^2} \right|_{S = S_J} = 0$, in the Euler scheme these mean:\\

\scalebox{1}{%
$\begin{aligned}
v(t_{i+1},y_l,S_0) = & ~ v(t_{i},y_l,S_0) \big [ 1 + \delta \big(\bar{\lambda}^a(y_l, S_0) + \bar{\lambda}^b(y_l, S_0)\big) \big] \, \\
&+ v(t_{i}, y_{l+1}, S_0) \big[-\bar{\lambda}^a(y_l, S_0) \big] \, \\
&+ v(t_{i}, y_{l-1}, S_0) \big[-\bar{\lambda}^b(y_l, S_0) \big] \, \\
&- \bar{\lambda}^a(y_l, S_0) \big[ \varphi_c \big(y_{l-1} \big) - \varphi_c \big(y_l \big) - \xi S_0 + r \big(\frac{c}{y_l^2} \big) \big] - \bar{\lambda}^b(y_l, S_0) \big[ \varphi_c \big(y_{l+1} \big) - \varphi_c \big(y_l \big) + \xi S_0 + r \big(\frac{c}{y_l^2} \big) \big] \
\end{aligned}$
}\\

\scalebox{1}{%
$\begin{aligned}
v(t_{i+1},y_l,S_J) = & ~ v(t_{i},y_l,S_J) \big [ 1 - \delta \big( \bar{\lambda}^a(y_l, S_J) + \bar{\lambda}^b(y_l, S_J)\big) \big] \, \\
&+ v(t_{i}, y_{l+1}, S_J) \big[-\bar{\lambda}^a(y_l, S_J) \big] \, \\
&+ v(t_{i}, y_{l-1}, S_J) \big[-\bar{\lambda}^b(y_l, S_J) \big] \, \\
&- \bar{\lambda}^a(y_l, S_J) \big[ \varphi_c \big(y_{l-1} \big) - \varphi_c \big(y_l \big) - \xi S_J + r \big(\frac{c}{y_l^2} \big) \big] - \bar{\lambda}^b(y_l, S_J) \big[ \varphi_c \big(y_{l+1} \big) - \varphi_c \big(y_l \big) + \xi S_J + r \big(\frac{c}{y_l^2} \big) \big] \
\end{aligned}$
}\\ \\

We denote:\\

$\begin{aligned}
& c_0(y_l,S_j) = \big [ 1 + \delta \big(\bar{\lambda}^a(y_l, S_j) + \bar{\lambda}^b(y_l, S_j)\big) \big] \\
&c_1(y_l,S_j) = -\frac{\delta \sigma^2}{2h^2} = c_1\\
&c_2(y_l,S_j) = \big [ 1 + \delta \big(\frac{\sigma^2}{h^2} + \bar{\lambda}^a(y_l, S_j) + \bar{\lambda}^b(y_l, S_j)\big) \big] \\
&c_3(y_l,S_j) = -\frac{\delta \sigma^2}{2h^2} = c_3\\
\end{aligned}$\\ \\

Linear system to solve: $V^{i+1,l} = A V^{i,l}$ for $i \in \{0,...,I-1\}$ where:\\

$A = \begin{bmatrix}
c_0(y_l,S_0) & 0 & 0 & \cdots &  & 0 \\
c_1 & c_2(y_l,S_1) & c_3 & \ddots &  & \vdots \\
0 & c_1 & c_2(y_l,S_2) & \ddots &  &  \\
\vdots & \ddots & \ddots & \ddots & c_3 & 0 \\
 &  &  & c_1 & c_2(y_l,S_{J-1}) & c_3 \\
0 & \cdots &  & 0 & 0 & c_0(y_l,S_J)
\end{bmatrix}
\in \mathbb{R}^{J+1,J+1}$, ~~~~~
$V^{i,l} =
\begin{bmatrix}
v(t_i, y_l, S_0) \\
v(t_i, y_l, S_1) \\
~ \\ \vdots \\ ~ \\
v(t_i, y_l, S_{J-1}) \\
v(t_i, y_l, S_J)
\end{bmatrix}
\in \mathbb{R}^{J+1}
$\\
